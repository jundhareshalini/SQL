ORDER BY COUNT(*) DESC;

--- COUNT OF EACH JOB_ID WHERE COUNT OF JOB_ID IS GREATER THAN 2 

SELECT JOB_ID, COUNT(*) FROM EMPLOYEES
GROUP BY JOB_ID
HAVING COUNT(*) > 2

-------- FETCH THE RECORDS WHERE SALARY IS GREATER THAN 20000

-- Find the total salary for each JOB_ID and DEPARTMENT_ID,
-- then display only those groups whose total salary is greater than 20000.

--- SYNTACTICALLY IS CORRECTLY BUT LOGICALLY INCORRECT -- GIVES WRONG OUTPUT

SELECT JOB_ID, DEPARTMENT_ID, SUM(SALARY) AS TOTAL_SALARY FROM EMPLOYEES
WHERE DEPARTMENT_ID IN (10,20,30,90) AND SALARY > 20000
GROUP BY JOB_ID, DEPARTMENT_ID
ORDER BY DEPARTMENT_ID;

----- CORRECT QUERY

SELECT JOB_ID, DEPARTMENT_ID, SUM(SALARY) AS TOTAL_SALARY FROM EMPLOYEES
WHERE DEPARTMENT_ID IN (10,20,30,90)
GROUP BY JOB_ID, DEPARTMENT_ID
HAVING SUM(SALARY) > 20000
ORDER BY DEPARTMENT_ID;
