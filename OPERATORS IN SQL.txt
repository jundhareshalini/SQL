---------------------- ARITHMETIC OPERATORS

SELECT * FROM EMPLOYEES;

--- I WANT TO DISPLAY EMPLOYEE_ID, FIRST_NAME,LAST_NAME, SALARY, BONUS - 10% BONUS BASED ON SALARY

SELECT EMPLOYEE_ID, 
FIRST_NAME,
LAST_NAME, 
SALARY, (SALARY * (10/100)) AS BONUS
FROM EMPLOYEES;

--- I WANT TO DISPLAY EMPLOYEE_ID, FIRST_NAME,LAST_NAME, SALARY, BONUS - 10% BONUS BASED ON SALARY AND TOTAL SALARY

    SELECT EMPLOYEE_ID, 
        FIRST_NAME,
        LAST_NAME, 
        SALARY, 
        SALARY * 0.1 AS BONUS,                                      ---- 10%
        --SALARY * 1.1 AS TOTAL_ANN_SALARY FROM EMPLOYEES;            ----- 1 IS 100% AND 1.1 IS 110%
        SALARY + (SALARY * 0.1) AS TOTAL_ANN_SALARY FROM EMPLOYEES;  
        

--- I WANT TO DISPLAY EMPLOYEE_ID, FIRST_NAME,LAST_NAME, SALARY AND TOTAL SALARY DECREASE BY 20%

    SELECT EMPLOYEE_ID, 
        FIRST_NAME,
        LAST_NAME, 
        SALARY,                                   
        --SALARY - (SALARY * 0.2) AS TOTAL_ANN_SALARY FROM EMPLOYEES;  
        SALARY * 0.80 AS TOTAL_ANN_SALARY FROM EMPLOYEES;      

--- I WANT TO DISPLAY EMPLOYEE_ID, FIRST_NAME,LAST_NAME, SALARY, BONUS(10% BONUS BASED ON SALARY) AND TOTAL SALARY, AND ACTUAL SALARY (SALARY - COMMISSION_PCT)

    SELECT EMPLOYEE_ID, 
        FIRST_NAME,
        LAST_NAME, 
        SALARY, 
        SALARY * 0.1 AS BONUS,                                     
        SALARY + (SALARY * 0.1) AS TOTAL_ANN_SALARY,
        SALARY + (SALARY * 0.1) - COMMISSION_PCT AS ACTUAL_SALARY
        FROM EMPLOYEES
        WHERE SALARY + (SALARY * 0.1) - COMMISSION_PCT IS NOT NULL;

---IN
SELECT EMPLOYEE_ID,
    FIRST_NAME,
    LAST_NAME,
    DEPARTMENT_ID
    FROM EMPLOYEES 
    WHERE DEPARTMENT_ID IN (30,90,100);
    
---NOT
SELECT EMPLOYEE_ID,
    FIRST_NAME,
    LAST_NAME,
    DEPARTMENT_ID
    FROM EMPLOYEES 
    WHERE DEPARTMENT_ID NOT IN (30,90,100);

---OR
SELECT EMPLOYEE_ID,
    FIRST_NAME,
    LAST_NAME,
    DEPARTMENT_ID
    FROM EMPLOYEES 
    WHERE FIRST_NAME = 'Steven' OR FIRST_NAME = 'Alexander';

---AND
SELECT EMPLOYEE_ID,
    FIRST_NAME,
    LAST_NAME,
    DEPARTMENT_ID
    FROM EMPLOYEES 
    WHERE SALARY > 20000 AND DEPARTMENT_ID IN (90,20,30);

---BETWEEN
SELECT EMPLOYEE_ID,
    FIRST_NAME,
    LAST_NAME,
    SALARY,
    DEPARTMENT_ID
    FROM EMPLOYEES 
    WHERE SALARY BETWEEN 10000 AND 20000;
    
----- LIKE
SELECT * FROM EMPLOYEES WHERE FIRST_NAME LIKE 'Steven';

---STARTS WITH 'S'
SELECT * FROM EMPLOYEES WHERE FIRST_NAME LIKE 'S%';  

---ENDS WITH 's'
SELECT * FROM EMPLOYEES WHERE FIRST_NAME LIKE '%s';  

---Displays all employee records whose first name contains the letter 'S' anywhere in the name.
SELECT * FROM EMPLOYEES WHERE FIRST_NAME LIKE '%S%';  

--DISPLAY ALL RECORDS WHERE FIRST_NAME HAS 'a' IN IT AND LAST_NAME ENDS WITH 's'
SELECT * FROM EMPLOYEES     
    WHERE FIRST_NAME LIKE '%a%'
    AND LAST_NAME LIKE '%s';  

--DISPLAY ALL RECORDS WHERE FIRST_NAME HAS 'a' IN IT AND LAST_NAME ENDS WITH 's' ANS SALARY > 5000 AND DEPARTMENT_ID IN (30,40)
SELECT * FROM EMPLOYEES 
    WHERE FIRST_NAME LIKE '%a%'
    AND LAST_NAME LIKE '%s'
    AND SALARY > 5000
    AND DEPARTMENT_ID IN (30,40);
